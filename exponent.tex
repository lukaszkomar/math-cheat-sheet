\begin{array}{lclclcl}  
  a^n\cdot a^m &=& a^{n+m}       &\qquad& \frac{a^n}{a^m}               &=& a^{n-m}\\
  (a\cdot b)^n &=& a^n \cdot b^n &\qquad& \left(\frac{a}{b}\right)^n    &=& \frac{a^n}{b^n}\\
  (a^n)^m      &=& a^{n \cdot m} &\qquad& \left(\frac{a}{b}\right)^{-n} &=& \left(\frac{b}{a}\right)^{n}\\
\end{array}
