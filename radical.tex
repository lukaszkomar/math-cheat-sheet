\begin{array}{lclclcl}  
  \sqrt[n]{a}           &=& a^{\frac{1}{n}}                 &\qquad& \sqrt[n]{a^m}         &=& a^{\frac{m}{n}} \\
  \sqrt[n]{\frac{a}{b}} &=& \frac{\sqrt[n]{a}}{\sqrt[n]{b}} &\qquad& \sqrt[n]{a\cdot b}    &=& \sqrt[n]{a}\cdot\sqrt[n]{b} \\
  \sqrt[n]{a^n}         &=& |a|                             &\qquad& \sqrt[m]{\sqrt[n]{a}} &=& \sqrt[m \cdot n]{a}\\  
\end{array}
\end{center}
