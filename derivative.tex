\begin{array}{lll}
 y=f\Big(g(x)\Big)   &\qquad& y' = f'(g) \cdot g'\\
 y=f(x) \cdot g(x)   &\qquad& y' = f' \cdot g + f \cdot g'\\  
 y=\frac{f(x)}{g(x)} &\qquad& y' = \frac{f' \cdot g - f \cdot g'}{g^2}\\  
\end{array}

\end{array}
